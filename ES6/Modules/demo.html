深入浅出ES6（十六）：模块 Modules 
// math.js
"use strict";
export {add as sum};
export var pi = 3.141593;;

function add (x, y) {
  return x + y;
};

// app.js
"use strict";
import * as math from "./math";
console.log("math.pi = " + math.pi); // math.pi = 3.141593

import {sum as add, pi} from "./math";
console.log("2π = " + add(pi, pi)); // 2π = 6.283186

// mathplusplus.js
"use strict";
export * from "./math";
export var e = 2.71828182846;
export default function(x) {
    return Math.exp(x);
}

// app2.js
"use strict";
import exp, {sum, pi, e} from "./mathplusplus";
console.log("e^π = " + exp(pi));    // e^π = 23.14070064895278
console.log("pi + e = " + sum(pi, e));  // pi + e = 5.85987482846
